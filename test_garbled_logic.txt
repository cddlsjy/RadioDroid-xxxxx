// 测试乱码检测逻辑
public class TestGarbledDetection {
    public static void main(String[] args) {
        System.out.println("Testing garbled character detection logic");
        
        // 测试1: 正常文本
        String normalText = "Normal text";
        System.out.println("\nTest 1: " + normalText);
        System.out.println("Should be detected as normal text");
        
        // 测试2: 包含欧洲字符的文本
        String europeanChars = "Cafe resume naive";
        System.out.println("\nTest 2: " + europeanChars);
        System.out.println("Should be detected as garbled text");
        
        // 测试3: 包含Box Drawing字符的文本
        String boxDrawing = "Song with box characters: ----";
        System.out.println("\nTest 3: " + boxDrawing);
        System.out.println("Should be detected as garbled text");
        
        // 测试4: 模拟Big5编码被误解析为UTF-8
        byte[] big5Bytes = {(byte)0xA7, (byte)0x41, (byte)0xA6, (byte)0x6E};
        String misinterpreted = new String(big5Bytes, java.nio.charset.StandardCharsets.UTF_8);
        System.out.println("\nTest 4: " + misinterpreted);
        System.out.println("Should be detected as garbled text");
        
        // 测试5: 混合文本
        String mixedText = "Normal text with some special chars";
        System.out.println("\nTest 5: " + mixedText);
        System.out.println("Should be detected based on the ratio of special characters");
        
        System.out.println("\nAll tests completed. The implementation in ExoPlayerWrapper.java should:");
        System.out.println("1. Detect obvious garbled characters using containsObviousGarbledCharacters()");
        System.out.println("2. Return '未知' when garbled text is detected");
        System.out.println("3. Apply this check at multiple points in the decoding process");
    }
}
